sudo nano /etc/nginx/sites-available/sistema.scopebrazil.com

server {
    listen 80;
    server_name sistema.scopebrazil.com;

    location / {
        proxy_pass http://127.0.0.1:5555;
        proxy_http_version 1.1;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_read_timeout 60s;
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
    }
}

sudo ln -s /etc/nginx/sites-available/sistema.scopebrazil.com /etc/nginx/sites-enabled/
sudo nginx -t
sudo systemctl reload nginx


sudo certbot --nginx -d sistema.scopebrazil.com
